&=\frac{count(X=x,Y=y) \cdot count(Y=y,Z=z)}{T \cdot count(Y=y)}

&=\frac{count(X=x,Y=y) \cdot count(Y=y,Z=z)}{T \cdot count(Y=y)}
\end{align*}
